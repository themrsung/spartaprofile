function save_reputation() {
    rep = $('#user-reputation').val();

    $.ajax({
        type: "POST",
        url: "/mjsungpostrep",
        data: {
            rep_give: rep
        },
        success: function (response) {
            show_reputation()
            $('#user-reputation').hide();
            $('#user-reputation-button').hide();
            $('#user-reputation-title').hide();
        }
    });
}
